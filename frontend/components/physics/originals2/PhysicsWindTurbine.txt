// Returns the constraint-based color (green / amber / red)
function stressColor(value, warn, crit) {
  if (value === undefined) return "#ffffff";
  if (value >= crit) return "#fca5a5";
  if (value >= warn) return "#fcd34d";
  return "#ffffff";
}
